def solution(participant, completion):
    answer = ''
    names = dict()
	for name in participant:
        names[name] = names.get(name, 0) + 1
    for name in completion:
        names[name] -= 1
    for name in names:
        if names[name] == 1:
            answer = name
            return answer
        
    return answer
