def solution(tickets):
    answer = ["ICN"]
    # λ„μ°©μ§€ κΈ°μ¤€ μ•νλ²³μ μ •λ ¬
    tickets = sorted(tickets,key= lambda x:x[1])
    while tickets:
      for ticket in tickets:
        if ticket[0] == answer[-1]:
          answer.append(ticket[1])
          tickets.remove(ticket)
          break
    
    return answer
