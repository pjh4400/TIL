from collections import deque


def solution(priorities, location):
    answer = 0
    q = deque([(v,i) for i,v in enumerate(priorities)])
    
    while q:
      p, idx = q.popleft()
      if q and max(q)[0] > p:
        q.append((p,idx))
      else:
        answer += 1
        if idx == location:
          break 

    return answer
