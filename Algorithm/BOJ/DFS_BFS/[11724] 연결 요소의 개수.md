```python
import sys
sys.setrecursionlimit(10000)
def dfs(v):
  visited[v] = True
  for i in graph[v]:
    if not visited[i]:
      dfs(i)

n, m = map(int,sys.stdin.readline().split())
graph = [[] for _ in range(n+1)]
for _ in range(m):
  u, v = map(int,sys.stdin.readline().split())
  graph[u].append(v)
  graph[v].append(u)

visited = [False] * (n+1)
count = 0
for i in range(1,n+1):
  if not visited[i]:
    dfs(i)
    count += 1

print(count)
```

```python
import sys
from collections import deque
def bfs():
  count = 0
  for i in range(1,n+1):
    if not visited[i]:
      count += 1
      queue = deque([i])
      visited[i] = True
      while queue:
        v = queue.popleft()
        for j in graph[v]:
          if not visited[j]:
            queue.append(j)
            visited[j] = True
  print(count)

n, m = map(int,sys.stdin.readline().split())
graph = [[] for _ in range(n+1)]
for _ in range(m):
  u, v = map(int,sys.stdin.readline().split())
  graph[u].append(v)
  graph[v].append(u)

visited = [False] * (n+1)
count = 0
bfs()
```
