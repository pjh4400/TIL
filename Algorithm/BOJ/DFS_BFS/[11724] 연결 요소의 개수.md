from collections import deque
def bfs():
  count = 0
  for i in range(1,n+1):
    if not visited[i]:
      count += 1
      queue = deque([i])
      visited[i] = True
      while queue:
        v = queue.popleft()
        for j in graph[v]:
          if not visited[j]:
            queue.append(j)
            visited[j] = True
  print(count)

n, m = map(int,input().split())
graph = [[] for _ in range(n+1)]
for _ in range(m):
  u, v = map(int,input().split())
  graph[u].append(v)
  graph[v].append(u)

count = 0
bfs()
